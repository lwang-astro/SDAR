37  void recursiveSymplecticCofficientsSplit(Float* _ck, const int _n) {
38  if (_n==1) {
39  _ck[0] = 0.5;
40  _ck[1] = 1.0;
41  _ck[2] = 0.5;
42  }
43  else if (_n>1) {
44  // order 2(n-1)
45  recursiveSymplecticCofficientsSplit(_ck, _n-1);
46 
47  // last order array size
48  const int n_size=std::pow(3, _n-1);
49 
50  // backup last order cofficients
51  Float ck_last[n_size];
52  for (int i=0; i<n_size; i++) ck_last[i] = _ck[i];
53 
54  // 2^(1/(2n-1))
55  Float ap = pow(2.0, 1.0/(2*_n-1));
56 
57  Float w[3];
58  w[0] = 1.0/(2.0-ap);
59  w[1] = - ap*w[0];
60  w[2] = w[0];
61 
62  for (int i=0; i<3; i++) {
63  for (int j=0; j<n_size; j++) {
64  _ck[i*n_size+j] = w[i]*ck_last[j];
65  }
66  }
67  }
68  else {
69  std::cerr<<"Error: integrator order should be positive, given value "<<_n<<std::endl;
70  abort();
71  }
