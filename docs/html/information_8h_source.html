130  inline Float calcDsElliptic(BinaryTree<Tparticle>& _bin, const Float& _G) {
131  //kepler orbit, step ds=dt*G*m1*m2/r estimation (1/32 orbit): 2*pi/32*sqrt(G*semi/(m1+m2))*m1*m2
